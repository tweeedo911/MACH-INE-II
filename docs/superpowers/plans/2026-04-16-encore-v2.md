# ENCORE v2 — Canon Machine Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Redesign the ENCORE track as a Canon Machine — 5 canonical voices playing transformations of the same generated phrase, escalation with compression, geometric block visuals, hard cut ending.

**Architecture:** A phrase generator in director3.js creates 7-13 note phrases on the active scale. The canon engine advances 5 voice positions at different speeds (1×, ½×, 2×, 3×, 1× offset). Each musical module reads its current note from `worldState.encoreCanon` instead of fixed patterns. Visual depositFns use geometric shapes (diagonals, quadrants, arcs, crosses) instead of scatter. The existing v1 encore code is replaced in-place.

**Tech Stack:** Vanilla JS ES modules, Canvas 2D, WebMIDI, Web Worker clock. Zero new dependencies.

**Spec:** `docs/superpowers/specs/2026-04-16-encore-v2-design.md`

// ── ENCORE v2: geometric deposit helpers ──

function depositDiagonal(field, angle, thickness, force) {
  // angle: 0 = top-left→bottom-right, 1 = top-right→bottom-left, 0.5 = vertical
  const W = _cellsX, H = _cellsY;
  const t = Math.max(1, thickness);
  for (let i = 0; i < H; i++) {
    // interpolate X position based on angle
    const baseX = angle < 0.5
      ? Math.floor(i * (W - 1) / (H - 1))           // ↘
      : W - 1 - Math.floor(i * (W - 1) / (H - 1));  // ↙
    for (let dx = -Math.floor(t/2); dx <= Math.floor(t/2); dx++) {
      const cx = baseX + dx;
      if (cx >= 0 && cx < W) depositPoint(field, cx, i, force);
    }
  }
}

function depositQuadrant(field, quadrant, fillRatio, force) {
  // quadrant: 0=TL, 1=TR, 2=BL, 3=BR. fillRatio: 0-1 (0=border only, 1=filled)
  const W = _cellsX, H = _cellsY;
  const hW = Math.floor(W / 2), hH = Math.floor(H / 2);
  const x0 = (quadrant & 1) ? hW : 0;
  const y0 = (quadrant & 2) ? hH : 0;
  const x1 = (quadrant & 1) ? W  : hW;
  const y1 = (quadrant & 2) ? H  : hH;
  for (let y = y0; y < y1; y++) {
    for (let x = x0; x < x1; x++) {
      // border always on; interior based on fillRatio
      const isBorder = (x === x0 || x === x1 - 1 || y === y0 || y === y1 - 1);
      if (isBorder || Math.random() < fillRatio) {
        depositPoint(field, x, y, force);
      }
    }
  }
}

function depositArc(field, radius, arcFraction, force) {
  // Draws an arc from center. radius in cells. arcFraction: 0.25=quarter, 1=full circle
  const W = _cellsX, H = _cellsY;
  const cx = Math.floor(W / 2), cy = Math.floor(H / 2);
  const r = Math.min(radius, Math.min(cx, cy));
  const steps = Math.max(12, Math.floor(r * 6.28 * arcFraction));
  const maxAngle = arcFraction * 6.2832;
  const startAngle = Math.random() * 6.2832;  // random rotation each call
  for (let i = 0; i < steps; i++) {
    const a = startAngle + (i / steps) * maxAngle;
    const px = cx + Math.round(r * Math.cos(a));
    const py = cy + Math.round(r * Math.sin(a) * (W / H));  // aspect correction
    if (px >= 0 && px < W && py >= 0 && py < H) {
      depositPoint(field, px, py, force);
    }
  }
}

function depositCross(field, cx, cy, armLen, force) {
  // + shape centered on (cx, cy)
  const W = _cellsX, H = _cellsY;
  for (let d = -armLen; d <= armLen; d++) {
    const px = cx + d, py = cy + d;
    if (px >= 0 && px < W) depositPoint(field, px, cy, force);  // horizontal
    if (py >= 0 && py < H) depositPoint(field, cx, py, force);  // vertical
  }
}

function depositHalf(field, top, fillCount, force) {
  // Fills top or bottom half of screen. fillCount = number of cells to light up.
  const W = _cellsX, H = _cellsY;
  const hH = Math.floor(H / 2);
  const y0 = top ? 0 : hH;
  const y1 = top ? hH : H;
  const maxCells = W * (y1 - y0);
  const n = Math.min(fillCount, maxCells);
  for (let i = 0; i < n; i++) {
    const px = Math.floor(Math.random() * W);
    const py = y0 + Math.floor(Math.random() * (y1 - y0));
    depositPoint(field, px, py, force);
  }
}

function depositFlash(fields, force) {
  // Full-screen flash on kick field — convergence event
  const W = _cellsX, H = _cellsY;
  for (let y = 0; y < H; y++) {
    for (let x = 0; x < W; x++) {
      fields.kick[y * W + x] = Math.min(1, fields.kick[y * W + x] + force);
    }
  }
}

// ── Phrase generator ──
function _generatePhrase(scale, len) {
  // Generates a phrase of `len` notes following contour rules (spec §2.1)
  const pool = scale.filter(n => n >= 48 && n <= 72);  // C3–C5 range for generation
  if (pool.length < 4) return pool.slice(0, len);

  const phrase = [];
  // Start on a note in the middle third of the pool
  const startIdx = Math.floor(pool.length / 3) + Math.floor(Math.random() * Math.floor(pool.length / 3));
  phrase.push(pool[startIdx]);

  // Decide overall direction: ascending (0.5 prob) or descending
  const ascending = Math.random() < 0.5;
  let leapUsed = false;
  let lastDirection = ascending ? 1 : -1;

  for (let i = 1; i < len; i++) {
    const lastNote = phrase[i - 1];
    const lastIdx = pool.indexOf(lastNote);
    if (lastIdx < 0) { phrase.push(pool[Math.floor(Math.random() * pool.length)]); continue; }

    const roll = Math.random();
    let interval, direction;

    if (roll < CFG.ENCORE_CONTOUR_STEP) {
      // Stepwise (2nd): 1 scale degree
      interval = 1;
      // Bias toward overall direction (70%) but allow reversal (30%)
      direction = Math.random() < 0.7 ? (ascending ? 1 : -1) : (ascending ? -1 : 1);
    } else if (roll < CFG.ENCORE_CONTOUR_STEP + CFG.ENCORE_CONTOUR_SKIP) {
      // Skip (3rd or 4th): 2-3 scale degrees, compensated
      interval = 2 + Math.floor(Math.random() * 2);
      direction = lastDirection;
      // Next note after this will go opposite (compensation happens naturally via direction bias)
    } else {
      // Leap (5th or 6th): 4-5 scale degrees, max once per phrase
      if (leapUsed) { interval = 1; direction = lastDirection; }
      else { interval = 4 + Math.floor(Math.random() * 2); direction = ascending ? 1 : -1; leapUsed = true; }
    }

    let newIdx = lastIdx + interval * direction;
    // Clamp to pool
    newIdx = Math.max(0, Math.min(pool.length - 1, newIdx));
    phrase.push(pool[newIdx]);
    lastDirection = direction;
  }

  // Closure: adjust last note to be root or 3rd/5th from first note
  const first = phrase[0];
  const closureTargets = pool.filter(n => {
    const interval = Math.abs(n - first) % 12;
    return interval === 0 || interval === 3 || interval === 4 || interval === 7 || interval === 8;
  });
  if (closureTargets.length > 0) {
    // Pick the closest closure target to the current last note
    const last = phrase[phrase.length - 1];
    closureTargets.sort((a, b) => Math.abs(a - last) - Math.abs(b - last));
    phrase[phrase.length - 1] = closureTargets[0];
  }

  return phrase;
}

// ── Phrase transformations ──
function _invertPhrase(phrase, pivotNote) {
  // Inversion: mirror intervals around pivot
  return phrase.map(n => pivotNote + (pivotNote - n));
}

function _retrogradePhrase(phrase) {
  return [...phrase].reverse();
}

// ── Transpose note into register ──
function _fitToRegister(note, lo, hi) {
  let n = note;
  while (n < lo) n += 12;
  while (n > hi) n -= 12;
  if (n < lo) n = lo;  // edge case
  return n;
}

// ── Convergence detection ──
function _checkConvergence(canon) {
  // Count how many active voices share the same pitch class
  const pitchClasses = {};
  const voices = ['bass', 'chord', 'arp', 'voice', 'lead'];
  let activeCount = 0;
  for (const v of voices) {
    if (!canon[v].active) continue;
    activeCount++;
    const pc = canon[v].note % 12;
    pitchClasses[pc] = (pitchClasses[pc] || 0) + 1;
  }
  if (activeCount < 3) return false;
  // Convergence if any pitch class has ≥3 voices
  for (const count of Object.values(pitchClasses)) {
    if (count >= 3) return true;
  }
  return false;
}
