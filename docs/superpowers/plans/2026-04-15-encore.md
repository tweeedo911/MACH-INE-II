# ENCORE Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an 8th track "ENCORE" with polimetric layers (6 cycle lengths from primes), 3 switchable octatonic/Prometheus scales, heartbeat intro, diamond-shaped form, and an extreme RGB visual biome.

**Architecture:** ENCORE is a self-contained track definition in tracks.js + biome in biomi.js. The key change is making musical modules support variable cycle lengths via `worldState.encoreCycleLen[module]`. Director3 manages the heartbeat→bricks→plateau→teardown state machine. Camera is fixed during encore.

**Tech Stack:** Vanilla JS ES modules, Canvas 2D, WebMIDI, Web Worker clock. Zero new dependencies.

**Spec:** `docs/superpowers/specs/2026-04-15-encore-design.md`

function _updateEncore(dt) {
  _phaseTime += dt;
  _totalTime += dt;
  
  const bpm = worldState.bpm || 132;
  const barDuration = 240 / bpm;
  
  _barAcc += dt;
  while (_barAcc >= barDuration) {
    _barAcc -= barDuration;
    _phaseBars++;
    _encoreBrickBar++;
    
    // Advance master clock
    // (rhythm.js handles globalTick — we just track bars here)
  }
  
  const brick = ENCORE_BRICKS[_encoreBrick];
  if (!brick) {
    // Past last brick — encore is done
    _endEncore();
    return;
  }
  
  // ── Heartbeat: BPM ramp ──
  if (_encoreBrick === 0) {
    // Exponential ramp: 60 → 132 over 8 bars
    const progress = Math.min(1, _encoreBrickBar / brick.bars);
    worldState.bpm = 60 * Math.pow(132 / 60, progress);
    
    // Velocity ramp: 40 → 110
    worldState.velocityCeiling.rhythm = Math.round(40 + progress * 70);
  }
  
  // ── Teardown: fade modules ──
  if (_encoreBrick === 8) {
    for (const td of ENCORE_TEARDOWN) {
      if (_encoreBrickBar >= td.bar && _encoreBrickBar < td.bar + 4) {
        // 4-bar fade: ceiling goes from current to 0
        const fadeProgress = (_encoreBrickBar - td.bar) / 4;
        const ceilMul = 1 - fadeProgress;
        if (td.module === 'melody') {
          worldState.velocityCeiling.melody = Math.round(90 * ceilMul);
        } else if (td.module === 'harmony') {
          worldState.velocityCeiling.harmony = Math.round(100 * ceilMul);
        } else if (td.module === 'bass') {
          worldState.velocityCeiling.bass = Math.round(110 * ceilMul);
          worldState.density.harmony = Math.max(0, 0.6 * ceilMul);  // drone fades with bass
        } else if (td.module === 'rhythm') {
          // Specific sub-role fading handled via density
          if (td.subRole === 'snare' || td.subRole === 'hat') {
            // Reduce rhythm density — snare/hat check density
            worldState.density.rhythm = Math.max(0.3, worldState.density.rhythm * ceilMul);
          }
        }
      }
      // Kick deceleration (bar 48-56)
      if (td.subRole === 'kick' && _encoreBrickBar >= td.bar) {
        const decelProgress = Math.min(1, (_encoreBrickBar - td.bar) / 8);
        worldState.bpm = 132 * Math.pow(60 / 132, decelProgress);
        worldState.velocityCeiling.rhythm = Math.round(110 * (1 - decelProgress));
      }
    }
  }
  
  // ── Brick transition ──
  if (_encoreBrickBar >= brick.bars) {
    _encoreBrick++;
    _encoreBrickBar = 0;
    
    const nextBrick = ENCORE_BRICKS[_encoreBrick];
    if (nextBrick) {
      // Apply module densities for new brick
      if (nextBrick.modules) {
        for (const [mod, val] of Object.entries(nextBrick.modules)) {
          worldState.density[mod] = val;
        }
        // Zero out modules not listed
        for (const mod of ['rhythm', 'harmony', 'bass', 'melody', 'texture']) {
          if (!(mod in nextBrick.modules)) worldState.density[mod] = 0;
        }
      }
      
      // After heartbeat: lock BPM at 132
      if (_encoreBrick >= 1 && worldState.bpm !== 132) {
        worldState.bpm = 132;
      }
      
      // Update phase label for visual coherence
      if (_encoreBrick <= 2) worldState.phase = 'pulsazione';
      else if (_encoreBrick <= 5) worldState.phase = 'densita';
      else if (_encoreBrick <= 7) worldState.phase = 'rottura';
      else worldState.phase = 'dissoluzione';
      
      worldState.camera.phase = worldState.phase;
      
      console.log(`[DIR3] ENCORE brick ${_encoreBrick}: ${nextBrick.name}`);
    } else {
      _endEncore();
    }
  }
}
